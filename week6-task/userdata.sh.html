#!/bin/bash # Update the system yum update -y # Install Apache Web Server yum install -y httpd # Start and enable Apache systemctl start httpd systemctl enable httpd # Get the Instance ID from the Metadata Service (IMDSv2) TOKEN=$(curl -X PUT "http://169.254.169.254/latest/api/token" -H "X-aws-ec2-metadata-token-ttl-seconds: 21600") INSTANCE_ID=$(curl -H "X-aws-ec2-metadata-token: $TOKEN" -s http://169.254.169.254/latest/meta-data/instance-id) # Create a simple webpage echo "

Hello from My Auto-Scaled App!

" > /var/www/html/index.html echo "

This request is being handled by Instance ID: $INSTANCE_ID

" >> /var/www/html/index.html echo "

High Availability in action!

" >> /var/www/html/index.html
